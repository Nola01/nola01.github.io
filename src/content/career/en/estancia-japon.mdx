<script is:inline set:html={`
  function setupCarousel() {
    const container = document.getElementById('carousel-japon');
    if (!container) return; 

    const items = container.querySelectorAll('.carousel-item');
    const prevBtn = container.querySelector('#prevBtn');
    const nextBtn = container.querySelector('#nextBtn');
    let currentIndex = 0;

    function showSlide(index) {
      items.forEach((item, i) => {
        if (i === index) {
          item.classList.remove('opacity-0', 'z-0');
          item.classList.add('opacity-100', 'z-10');
        } else {
          item.classList.remove('opacity-100', 'z-10');
          item.classList.add('opacity-0', 'z-0');
        }
      });
    }

    if (nextBtn) {
      nextBtn.onclick = () => {
        currentIndex = (currentIndex + 1) % items.length;
        showSlide(currentIndex);
      };
    }

    if (prevBtn) {
      prevBtn.onclick = () => {
        currentIndex = (currentIndex - 1 + items.length) % items.length;
        showSlide(currentIndex);
      };
    }
  }

  setupCarousel();

  // Asegura que funcione con View Transitions de Astro
  document.addEventListener('astro:after-swap', setupCarousel);
`} />
